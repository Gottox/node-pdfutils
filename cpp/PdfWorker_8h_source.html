31  static void handleIntermediate(uv_async_t *handle UV_ASYNC_STATUS) {
32  PdfWorker *self = (PdfWorker *)handle->data;
33  void *data;
34 
35  while(true) {
36  uv_mutex_lock(&self->mutex);
37  if(self->intermediate.empty()) {
38  uv_mutex_unlock(&self->mutex);
39  break;
40  } else {
41  data = self->intermediate.front();
42  self->intermediate.pop_front();
43  uv_mutex_unlock(&self->mutex);
44  }
45  self->HandleIntermediate(data);
46  }
47  }
